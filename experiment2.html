Generating Prime Numbers Between 1 and 100

Problem Statement

Generate all prime numbers between 1 and 100 using a JavaScript function.

Code Implementation


function isPrime(num) {
    if (num <= 1) return false;
    for (let i = 2; i < num; i++) {
        if (num % i === 0) return false;
    }
    return true;
}

function generatePrimes() {
    let primes = [];
    for (let i = 2; i <= 100; i++) {
        if (isPrime(i)) {
            primes.push(i);
        }
    }
    return primes;
}

console.log(generatePrimes());

Result

The function successfully generates a list of prime numbers between 1 and 100.
